ajax异步上传文件（表单形式）
$("#upload-file").on("change", function() {
	$.ajax({
		url: 'UploadImgServlet',
		type: 'POST',
		cache: false,
		data: new FormData($('#upload-head-img')[0]), //h5的DataForm对象
		dataType: "json",
		processData: false,
		contentType: false
	})
})
